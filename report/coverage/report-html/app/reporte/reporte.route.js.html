    .module('app.reporte')
    .run(appRun);
 
  appRun.$inject = ['routerHelper', 'Auth'];

  function appRun(routerHelper, Auth) {
    routerHelper.configureStates(getStates());
  }
 
  function getStates() {
    return [{
      state: 'reporte',
      config: {
        abstract: true,
        url: '/reporte',
      }
    }, {
      state: 'reporte.feed',

        controller: 'ReporteFeedController',
